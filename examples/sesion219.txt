(defn db->map [db]
  (map (juxt identity #(% db))
        (keys (h/delegate db))))

(defn schema-info [db]
  (d/q '{:find [[(pull ?id [* {:db/valueType [:db/ident]}
                              {:db/cardinality [:db/ident]}
                              {:db/unique [:db/ident]}]) ...]]
          :where [[?e :db/ident ?id]
                  [_ :db.install/attribute ?e]]}
        db))
